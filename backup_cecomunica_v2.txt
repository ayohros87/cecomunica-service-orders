@echo off
setlocal enabledelayedexpansion

:: === Fecha y hora para nombre del backup ===
for /f %%i in ('powershell -command "Get-Date -Format yyyy-MM-dd_HH-mm"') do set datetime=%%i
set folderName=backup_%datetime%

:: Carpeta destino de backups
set backupDir=%cd%\backups
if not exist "%backupDir%" mkdir "%backupDir%"

:: Carpeta temporal para armar el backup
set fullPath=%backupDir%\%folderName%
set zipPath=%backupDir%\%folderName%.zip

echo Creando carpeta temporal %fullPath% ...
mkdir "%fullPath%"

:: === Copia raíz del proyecto ===
copy /Y *.html "%fullPath%" >nul 2>&1
copy /Y *.css "%fullPath%" >nul 2>&1
copy /Y *.js "%fullPath%" >nul 2>&1
copy /Y *.json "%fullPath%" >nul 2>&1
copy /Y *.txt "%fullPath%" >nul 2>&1
copy /Y *.md "%fullPath%" >nul 2>&1
copy /Y firebase.json "%fullPath%" >nul 2>&1
copy /Y .firebaserc "%fullPath%" >nul 2>&1
copy /Y firestore.rules "%fullPath%" >nul 2>&1
copy /Y firestore.indexes.json "%fullPath%" >nul 2>&1
copy /Y package*.json "%fullPath%" >nul 2>&1

:: === Copia carpeta public completa ===
xcopy public "%fullPath%\public" /E /I /Y >nul 2>&1

:: Extra: imágenes y fuentes
xcopy public\img "%fullPath%\public\img" /E /I /Y >nul 2>&1
xcopy public\fonts "%fullPath%\public\fonts" /E /I /Y >nul 2>&1

:: === Copia carpetas de módulos ===
xcopy ordenes "%fullPath%\ordenes" /E /I /Y >nul 2>&1
xcopy POC "%fullPath%\POC" /E /I /Y >nul 2>&1
xcopy inventario "%fullPath%\inventario" /E /I /Y >nul 2>&1
xcopy contratos "%fullPath%\contratos" /E /I /Y >nul 2>&1

:: === Crear archivo ZIP con PowerShell ===
echo Comprimiendo backup en %zipPath% ...
powershell -command "Compress-Archive -Path '%fullPath%\*' -DestinationPath '%zipPath%' -Force"

:: === Eliminar carpeta temporal ===
rmdir /s /q "%fullPath%"

echo.
echo ✅ Backup completo creado: %zipPath%
pause
